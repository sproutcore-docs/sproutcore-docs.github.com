  8 var MyFoo = null, callInfo ;
  9 module("SC.Record#refresh", {
 10   setup: function() {
 11     SC.RunLoop.begin();
 12     MyApp = SC.Object.create({
 13       store: SC.Store.create()
 14     })  ;
 15   
 16     MyApp.Foo = SC.Record.extend();
 17     MyApp.json = { 
 18       foo: "bar", 
 19       number: 123,
 20       bool: YES,
 21       array: [1,2,3] 
 22     };
 23     
 24     MyApp.foo = MyApp.store.createRecord(MyApp.Foo, MyApp.json);
 25     
 26     // modify store so that everytime refreshRecords() is called it updates 
 27     // callInfo
 28     callInfo = null ;
 29     MyApp.store.refreshRecord = function(records) {
 30       callInfo = SC.A(arguments) ; // save method call
 31     };
 32   },
 33   
 34   teardown: function() {
 35     SC.RunLoop.end();
 36   }
 37   
 38 });
 39 
 40 test("calling refresh should call refreshRecord() on store", function() {
 41   MyApp.foo.refresh();
 42   same(callInfo, [null,null,MyApp.foo.storeKey,undefined], 'refreshRecord() should be called on parent');
 43 });
 44 
 45 test("should return receiver", function() {
 46   equals(MyApp.foo.refresh(), MyApp.foo, 'should return receiver');
 47 });
