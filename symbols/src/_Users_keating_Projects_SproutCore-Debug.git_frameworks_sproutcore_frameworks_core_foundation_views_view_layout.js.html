108   /**
109     This convenience method will take the current layout, apply any changes
110     you pass and set it again.  It is more convenient than having to do this
111     yourself sometimes.
112 
113     You can pass just a key/value pair or a hash with several pairs.  You can
114     also pass a null value to delete a property.
115 
116     This method will avoid actually setting the layout if the value you pass
117     does not edit the layout.
118 
119     @param {String|Hash} key
120     @param {Object} value
121     @returns {SC.View} receiver
122   */
123   adjust: function(key, value) {
124     var layout = this.get('layout'), didChange = NO, cur, hash;
125 
126     if (key === undefined) { return this ; } // nothing to do.
127 
128     // handle string case
129     if (SC.typeOf(key) === SC.T_STRING) {
130       // this is copied from below
131       cur = layout[key];
132 
133       if(value === undefined || cur == value) return this;
134 
135       layout = SC.clone(layout);
136 
137       if(value === null) {
138         delete layout[key];
139       } else {
140         layout[key] = value;
141       }
142 
143       didChange = YES;
144     }
145 
146     else {
147       hash = key;
148 
149       for(key in hash) {
150         if (!hash.hasOwnProperty(key)) { continue; }
151 
152         value = hash[key] ;
153         cur = layout[key] ;
154 
155         if (value === undefined || cur == value) { continue; }
156 
157         // only clone the layout the first time we see a change
158         if(!didChange) layout = SC.clone(layout);
159 
160         if (value === null) {
161           delete layout[key] ;
162         } else {
163           layout[key] = value ;
164         }
165 
166         didChange = YES;
167       }
168     }
169 
170     // now set adjusted layout
171     if (didChange) {
172       this.set('layout', layout) ;
173     }
174 
175     return this ;

  /**
371     Computes what the frame of this view would be if the parent were resized
372     to the passed dimensions.  You can use this method to project the size of
373     a frame based on the resize behavior of the parent.
374 
375     This method is used especially by the scroll view to automatically
376     calculate when scrollviews should be visible.
377 
378     Passing null for the parent dimensions will use the actual current
379     parent dimensions.  This is the same method used to calculate the current
380     frame when it changes.
381 
382     @param {Rect} pdim the projected parent dimensions
383     @returns {Rect} the computed frame
384   */
385   computeFrameWithParentFrame: function(original, pdim) {
386     var f, layout = this.get('layout');
387 
388     // We can't predict the frame for static layout, so just return the view's
389     // current frame (see original computeFrameWithParentFrame in views/view.js)
390     if (this.get('useStaticLayout')) {
391       f = original();
392       return f ? this._adjustForBorder(f, layout) : null;
393     } else {
394       f = {};
395     }
396 
397     var error, layer, AUTO = SC.LAYOUT_AUTO,
398         pv = this.get('parentView'),
399         dH, dW, //shortHand for parentDimensions
400         lR = layout.right,
401         lL = layout.left,
402         lT = layout.top,
403         lB = layout.bottom,
404         lW = layout.width,
405         lH = layout.height,
406         lcX = layout.centerX,
407         lcY = layout.centerY;
408 
409     if (lW === AUTO) {
410       error = SC.Error.desc(("%@.layout() cannot use width:auto if "+
411                 "staticLayout is disabled").fmt(this), "%@".fmt(this), -1);
412       SC.Logger.error(error.toString()) ;
413       throw error ;
414     }
415 
416     if (lH === AUTO) {
417        error = SC.Error.desc(("%@.layout() cannot use height:auto if "+
418                 "staticLayout is disabled").fmt(this),"%@".fmt(this), -1);
419        SC.Logger.error(error.toString())  ;
420       throw error ;
421     }
422 
423     if (!pdim) { pdim = this.computeParentDimensions(layout) ; }
424     dH = pdim.height;
425     dW = pdim.width;
426 
427     // handle left aligned and left/right
428     if (!SC.none(lL)) {
429       if(SC.isPercentage(lL)){
430         f.x = dW*lL;
431       }else{
432         f.x = lL ;
433       }
434       if (lW !== undefined) {
435         if(lW === AUTO) { f.width = AUTO ; }
436         else if(SC.isPercentage(lW)) { f.width = dW*lW ; }
437         else { f.width = lW ; }
438       } else { // better have lR!
439         f.width = dW - f.x ;
440         if(lR && SC.isPercentage(lR)) { f.width = f.width - (lR*dW) ; }
441         else { f.width = f.width - (lR || 0) ; }
442       }
443     // handle right aligned
444     } else if (!SC.none(lR)) {
445       if (SC.none(lW)) {
446         if (SC.isPercentage(lR)) {
447           f.width = dW - (dW*lR) ;
448         }
449         else f.width = dW - lR ;
450         f.x = 0 ;
451       } else {
452         if(lW === AUTO) f.width = AUTO ;
453         else if(SC.isPercentage(lW)) f.width = dW*lW ;
454         else f.width = (lW || 0) ;
455         if (SC.isPercentage(lW)) f.x = dW - (lR*dW) - f.width ;
456         else f.x = dW - lR - f.width ;
457       }
458 
459     // handle centered
460     } else if (!SC.none(lcX)) {
461       if(lW === AUTO) f.width = AUTO ;
462       else if (SC.isPercentage(lW)) f.width = lW*dW ;
463       else f.width = (lW || 0) ;
464       if(SC.isPercentage(lcX)) f.x = (dW - f.width)/2 + (lcX*dW) ;
465       else f.x = (dW - f.width)/2 + lcX ;
466     } else {
467       f.x = 0 ; // fallback
468       if (SC.none(lW)) {
469         f.width = dW ;
470       } else {
471         if(lW === AUTO) f.width = AUTO ;
472         if (SC.isPercentage(lW)) f.width = lW*dW ;
473         else f.width = (lW || 0) ;
474       }
475     }
476 
477     // handle top aligned and top/bottom
478     if (!SC.none(lT)) {
479       if(SC.isPercentage(lT)) f.y = lT*dH ;
480       else f.y = lT ;
481       if (lH !== undefined) {
482         if(lH === AUTO) f.height = AUTO ;
483         else if(SC.isPercentage(lH)) f.height = lH*dH ;
484         else f.height = lH ;
485       } else { // better have lB!
486         if(lB && SC.isPercentage(lB)) f.height = dH - f.y - (lB*dH) ;
487         else f.height = dH - f.y - (lB || 0) ;
488       }
489 
490     // handle bottom aligned
491     } else if (!SC.none(lB)) {
492       if (SC.none(lH)) {
493         if (SC.isPercentage(lB)) f.height = dH - (lB*dH) ;
494         else f.height = dH - lB ;
495         f.y = 0 ;
496       } else {
497         if(lH === AUTO) f.height = AUTO ;
498         if (lH && SC.isPercentage(lH)) f.height = lH*dH ;
499         else f.height = (lH || 0) ;
500         if (SC.isPercentage(lB)) f.y = dH - (lB*dH) - f.height ;
501         else f.y = dH - lB - f.height ;
502       }
503 
504     // handle centered
505     } else if (!SC.none(lcY)) {
506       if(lH === AUTO) f.height = AUTO ;
507       if (lH && SC.isPercentage(lH)) f.height = lH*dH ;
508       else f.height = (lH || 0) ;
509       if (SC.isPercentage(lcY)) f.y = (dH - f.height)/2 + (lcY*dH) ;
510       else f.y = (dH - f.height)/2 + lcY ;
511 
512     // fallback
513     } else {
514       f.y = 0 ; // fallback
515       if (SC.none(lH)) {
516         f.height = dH ;
517       } else {
518         if(lH === AUTO) f.height = AUTO ;
519         if (SC.isPercentage(lH)) f.height = lH*dH ;
520         else f.height = lH || 0 ;
521       }
522     }
523 
524     f.x = Math.floor(f.x);
525     f.y = Math.floor(f.y);
526     if(f.height !== AUTO) f.height = Math.floor(f.height);
527     if(f.width !== AUTO) f.width = Math.floor(f.width);
528 
529     // if width or height were set to auto and we have a layer, try lookup
530     if (f.height === AUTO || f.width === AUTO) {
531       layer = this.get('layer');
532       if (f.height === AUTO) f.height = layer ? layer.clientHeight : 0;
533       if (f.width === AUTO) f.width = layer ? layer.clientWidth : 0;
534     }
535 
536     f = this._adjustForBorder(f, layout);
537 
538     // Account for special cases inside ScrollView, where we adjust the
539     // element's scrollTop/scrollLeft property for performance reasons.
540     if (pv && pv.isScrollContainer) {
541       pv = pv.get('parentView');
542       f.x -= pv.get('horizontalScrollOffset');
543       f.y -= pv.get('verticalScrollOffset');
544     }
545 
546     // make sure the width/height fix min/max...
547     if (!SC.none(layout.maxHeight) && (f.height > layout.maxHeight)) {
548       f.height = layout.maxHeight ;
549     }
550 
551     if (!SC.none(layout.minHeight) && (f.height < layout.minHeight)) {
552       f.height = layout.minHeight ;
553     }
554 
555     if (!SC.none(layout.maxWidth) && (f.width > layout.maxWidth)) {
556       f.width = layout.maxWidth ;
557     }
558 
559     if (!SC.none(layout.minWidth) && (f.width < layout.minWidth)) {
560       f.width = layout.minWidth ;
561     }
562 
563     // make sure width/height are never < 0
564     if (f.height < 0) f.height = 0 ;
565     if (f.width < 0) f.width = 0 ;
566 
567     return f;

697     if (this.willBeginLiveResize) this.willBeginLiveResize() ;

701     for (idx=0; idx<len; ++idx) {

737   /**
738     This method is called whenever a property changes that invalidates the
739     layout of the view.  Changing the layout will do this automatically, but
740     you can add others if you want.
741 
742     Implementation Note:  In a traditional setup, we would simply observe
743     'layout' here, but as described above in the documentation for our custom
744     implementation of propertyDidChange(), this method must always run
745     immediately after 'layout' is updated to avoid the potential for stale
746     (incorrect) cached 'frame' values.
747 
748     @returns {SC.View} receiver
749   */
750   layoutDidChange: function() {
751     // Did our layout change in a way that could cause us to be resized?  If
752     // not, then there's no need to invalidate the frames of our child views.
753     var previousLayout = this._previousLayout,
754         currentLayout  = this.get('layout'),
755         didResize      = YES,
756         previousWidth, previousHeight, currentWidth, currentHeight;
757 
758 
759     // Handle old style rotation
760     if (!SC.none(currentLayout.rotate)) {
761       if (SC.none(currentLayout.rotateX)) {
762         currentLayout.rotateX = currentLayout.rotate;
763         SC.Logger.warn('Please set rotateX instead of rotate');
764       }
765     }
766     if (!SC.none(currentLayout.rotateX)) {
767       currentLayout.rotate = currentLayout.rotateX;
768     } else {
769       delete currentLayout.rotate;
770     }
771 
772     var animations = currentLayout.animations;
773     if (animations) {
774       if (!SC.none(animations.rotate)) {
775         if (SC.none(animations.rotateX)) {
776           animations.rotateX = animations.rotate;
777           SC.Logger.warn('Please animate rotateX instead of rotate');
778         }
779       }
780       if (!SC.none(animations.rotateX)) {
781         animations.rotate = animations.rotateX;
782       } else {
783         delete animations.rotate;
784       }
785     }
786 
787     if (previousLayout  &&  previousLayout !== currentLayout) {
788       // This is a simple check to see whether we think the view may have
789       // resized.  We could look for a number of cases, but for now we'll
790       // handle only one simple case:  if the width and height are both
791       // specified, and they have not changed.
792       previousWidth = previousLayout.width;
793       if (previousWidth !== undefined) {
794         currentWidth = currentLayout.width;
795         if (previousWidth === currentWidth) {
796           previousHeight = previousLayout.height;
797           if (previousLayout !== undefined) {
798             currentHeight = currentLayout.height;
799             if (previousHeight === currentHeight) didResize = NO;
800           }
801         }
802       }
803     }
804 
805     this.beginPropertyChanges() ;
806     this.notifyPropertyChange('hasAcceleratedLayer');
807     this.notifyPropertyChange('layoutStyle') ;
808     if (didResize) {
809       this.viewDidResize();
810     }
811     else {
812       // Even if we didn't resize, our frame might have changed.
813       // viewDidResize() handles this in the other case.
814       this._viewFrameDidChange();
815     }
816     this.endPropertyChanges() ;
817 
818     // notify layoutView...
819     var layoutView = this.get('layoutView');
820     if (layoutView) {
821       layoutView.set('childViewsNeedLayout', YES);
822       layoutView.layoutDidChangeFor(this) ;
823       if (layoutView.get('childViewsNeedLayout')) {
824         layoutView.invokeOnce(layoutView.layoutChildViewsIfNeeded);
825       }
826     }
827 
828     this._previousLayout = currentLayout;
829 
830     return this ;
831   },

969   convertLayoutToAnchoredLayout: function(layout, parentFrame){
970     var ret = {top: 0, left: 0, width: parentFrame.width, height: parentFrame.height},
971         pFW = parentFrame.width, pFH = parentFrame.height, //shortHand for parentDimensions
972         lR = layout.right,
973         lL = layout.left,
974         lT = layout.top,
975         lB = layout.bottom,
976         lW = layout.width,
977         lH = layout.height,
978         lcX = layout.centerX,
979         lcY = layout.centerY;
980 
981     // X Conversion
982     // handle left aligned and left/right
983     if (!SC.none(lL)) {
984       if(SC.isPercentage(lL)) ret.left = lL*pFW;
985       else ret.left = lL;
986       if (lW !== undefined) {
987         if(lW === SC.LAYOUT_AUTO) ret.width = SC.LAYOUT_AUTO ;
988         else if(SC.isPercentage(lW)) ret.width = lW*pFW ;
989         else ret.width = lW ;
990       } else {
991         if (lR && SC.isPercentage(lR)) ret.width = pFW - ret.left - (lR*pFW);
992         else ret.width = pFW - ret.left - (lR || 0);
993       }
994 
995     // handle right aligned
996     } else if (!SC.none(lR)) {
997 
998       // if no width, calculate it from the parent frame
999       if (SC.none(lW)) {
1000         ret.left = 0;
1001         if(lR && SC.isPercentage(lR)) ret.width = pFW - (lR*pFW);
1002         else ret.width = pFW - (lR || 0);
1003 
1004       // If has width, calculate the left anchor from the width and right and parent frame
1005       } else {
1006         if(lW === SC.LAYOUT_AUTO) ret.width = SC.LAYOUT_AUTO ;
1007         else {
1008           if (SC.isPercentage(lW)) ret.width = lW*pFW;
1009           else ret.width = lW;
1010           if (SC.isPercentage(lR)) ret.left = pFW - (ret.width + lR);
1011           else ret.left = pFW - (ret.width + lR);
1012         }
1013       }
1014 
1015     // handle centered
1016     } else if (!SC.none(lcX)) {
1017       if(lW && SC.isPercentage(lW)) ret.width = (lW*pFW) ;
1018       else ret.width = (lW || 0) ;
1019       ret.left = ((pFW - ret.width)/2);
1020       if (SC.isPercentage(lcX)) ret.left = ret.left + lcX*pFW;
1021       else ret.left = ret.left + lcX;

1024     } else if (!SC.none(lW)) {
1025       ret.left =  0;

1029         else ret.width = lW;

1040     if (layout.maxWidth !== undefined) ret.maxWidth = layout.maxWidth ;
1041 
1042     // Y Conversion
1043     // handle left aligned and top/bottom
1044     if (!SC.none(lT)) {
1045       if(SC.isPercentage(lT)) ret.top = lT*pFH;
1046       else ret.top = lT;

1054         else ret.height = ret.height - (lB || 0);

1064         else ret.height = pFH - (lB || 0);
