 10 module("SC.CollectionView.content", {
 11   setup: function() {
 12 
 13     // stub in collection view to verify that proper method are called
 14     view = SC.CollectionView.create({
 15 
 16       // ..........................................................
 17       // STUB: contentPropertyDidChange
 18       //
 19       contentPropertyDidChange: CoreTest.stub('contentPropertyDidChange'),
 20 
 21       // ..........................................................
 22       // STUB: computeLayout
 23       //
 24       computeLayout: CoreTest.stub('computeLayout', SC.CollectionView.prototype.computeLayout),
 25 
 26       // ..........................................................
 27       // STUB: RELOAD
 28       //
 29       reload: CoreTest.stub("reload", {
 30 
 31         // detect if we would reload everything.
 32         shouldReloadAll: function() {
 33           var history = this.history,
 34               loc = history.length,
 35               args;
 36 
 37           while(--loc >= 0) {
 38             args = history[loc];
 39             if (args <= 1) return YES ;
 40             if (args[1] === null) return YES ;
 41             if (args[0].get('nowShowing').contains(args[1])) return YES ;
 42           }
 43           return NO ;
 44         },
 45 
 46         // join all reload indexes passed excluding null or undefined
 47         indexes: function() {
 48           var history = this.history,
 49               loc = history.length,
 50               ret = SC.IndexSet.create(),
 51               args;
 52 
 53           while(--loc >= 0) {
 54             args = history[loc];
 55             if (args[1] && args[1].isIndexSet) ret.add(args[1]);
 56           }
 57 
 58           return ret ;
 59         },
 60 
 61         // need to save the passed indexes set as a clone because it may be
 62         // reused later...
 63         action: function(indexes) {
 64           var history = this.reload.history;  // note "this" is view
 65           if (indexes) {
 66             history[history.length-1][1] = indexes.clone();
 67           }
 68 
 69           // simulate calling computeLayout() to match original impl.
 70           this.computeLayout();
 71 
 72           return this ;
 73         },
 74 
 75         expect: function(indexes, callCount) {
 76 
 77           if (indexes === YES) {
 78             equals(this.shouldReloadAll(), YES, 'reload() should reload all');
 79           } else if (indexes !== NO) {
 80             var expected = this.indexes();
 81             var passed = indexes.isEqual(expected);
 82             ok(passed, "expected reload(%@), actual: reload(%@)".fmt(indexes, expected));
 83           }
 84 
 85           if (callCount !== undefined) {
 86             equals(this.callCount, callCount, 'reload() should have been called X times');
 87           }
 88 
 89           this.reset();
 90         }
 91 
 92       }),
 93 
 94       expectLength: function(len) {
 95         equals(view.get('length'), len, 'view.length after change');
 96 
 97 
 98         var nowShowing = view.get('nowShowing'),
 99             expected = SC.IndexSet.create(0,len);
100         ok(expected.isEqual(nowShowing), 'nowShowing expected: %@, actual: %@'.fmt(expected, nowShowing));
101       },
102 
103       // reset stubs
104       reset: function() {
105         this.reload.reset();
106         this.contentPropertyDidChange.reset();
107         this.computeLayout.reset();
108       }
109 
110     });
111 
112     content1 = "a b c d e f".w().map(function(x) {
113       return SC.Object.create({ title: x });
114     });
115 
116     content2 = "d e f x y z".w().map(function(x) {
117       return SC.Object.create({ title: x });
118     });
119 
120   }
121 });

127 test("setting content for first time", function() {
128   equals(view.get('content'), null, 'precond - view.content should be null');
129 
130   SC.run(function () {
131     view.set('content', content1);
132   });
133   view.reload.expect(YES, 2); // should reload everything
134   view.contentPropertyDidChange.expect(0); // should not call
135   view.computeLayout.expect(YES);
136   view.expectLength(content1.get('length'));
137 });
