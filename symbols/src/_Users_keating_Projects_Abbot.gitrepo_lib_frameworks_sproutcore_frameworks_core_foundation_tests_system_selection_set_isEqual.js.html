  9 var set1, set2, content;
 10 module("SC.SelectionSet.isEqual", {
 11   setup: function() {
 12     content = '1 2 3 4 5 6 7 8 9 0'.w();
 13     set1 = SC.SelectionSet.create();
 14     set2 = SC.SelectionSet.create();
 15   }
 16 });
 17 
 18 test("set.isEqual(same instance)", function() {
 19   ok(set1.isEqual(set1), 'same instance should return YES');
 20 });
 21 
 22 test("set.isEqual(null)", function() {
 23   ok(!set1.isEqual(null), 'null should return NO');
 24 });
 25 
 26 
 27 test("set1.isEqual(set2)", function() {
 28   ok(set1.isEqual(set2), 'same content should return YES');
 29   
 30   set1.add(content, 4,4);
 31   set2.add(content, 4,4);
 32   ok(set1.isEqual(set2), 'same content should return YES');
 33 
 34   set1.remove(content, 6);
 35   set2.remove(content, 6);
 36   ok(set1.isEqual(set2), 'same content should return YES');
 37 
 38   set1.remove(content, 4,4);
 39   set2.remove(content, 4,4);
 40   ok(set1.isEqual(set2), 'same content should return YES');
 41   
 42 });
 43 
 44 test("multiple content objects", function() {
 45   var content2 = "1 2 3 4 5".w();
 46   set1.add(content, 4,4).add(content2, 3);
 47   ok(!set1.isEqual(set2), 'should not be same when set2 is empty');
 48 
 49   set2.add(content2, 3);
 50   ok(!set1.isEqual(set2), 'should not be same when set2 has only one content');
 51 
 52   set2.add(content,4,4);
 53   ok(set1.isEqual(set2), 'should not be same when set2 has both content');
 54   
 55  56 });
 57 
 58 test("set1.isEqual(set2) after set2 is filled and emptied", function() {
 59   set2.add(content,4,4).remove(content,4,4);
 60   ok(set1.isEqual(set2), 'same content should return YES');
 61 });
