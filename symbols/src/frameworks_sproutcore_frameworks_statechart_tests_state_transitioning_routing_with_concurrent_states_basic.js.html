  6 var statechart, del, monitor, stateFoo, stateBar, stateA, stateB, stateC, stateX, stateY, TestState;
  7 
  8 module("SC.Statechart: Concurrent States - Trigger Routing on States Basic Tests", {
  9 
 10   setup: function() {
 11 
 12     del = SC.Object.create(SC.StatechartDelegate, {
 13 
 14       location: null,
 15 
 16       handlers: {},
 17 
 18       statechartUpdateLocationForState: function(statechart, location, state) {
 19         this.set('location', location);
 20       },
 21 
 22       statechartAcquireLocationForState: function(statechart, state) {
 23         return this.get('location');
 24       },
 25 
 26       statechartBindStateToRoute: function(statechart, state, route, handler) {
 27         this.handlers[route] = {
 28           statechart: statechart,
 29           state: state,
 30           handler: handler
 31         };
 32       }
 33 
 34     });
 35 
 36     TestState = SC.State.extend({
 37 
 38       enterState: function(context) {
 39         this.info = {};
 40         this.info.enterState = {
 41           state: this,
 42           context: context
 43         };
 44       }
 45 
 46     });
 47 
 48     statechart = SC.Statechart.create({
 49 
 50       monitorIsActive: YES,
 51 
 52       delegate: del,
 53 
 54       statesAreConcurrent: YES,
 55 
 56       foo: SC.State.design({
 57 
 58         initialSubstate: 'a',
 59 
 60         a: TestState.design({
 61 
 62           representRoute: 'dog'
 63 
 64         }),
 65 
 66         b: TestState.design({
 67 
 68           representRoute: 'cat'
 69 
 70         }),
 71 
 72         c: TestState.design({
 73 
 74           representRoute: ''
 75 
 76         })
 77 
 78       }),
 79 
 80       bar: SC.State.design({
 81 
 82         initialSubstate: 'x',
 83 
 84         x: TestState.design({
 85 
 86           representRoute: 'pig'
 87 
 88         }),
 89 
 90         y: TestState.design({
 91 
 92           representRoute: 'cow',
 93 
 94           enterStateByRoute: function(context) {
 95             this.info = {};
 96             this.info.enterStateByRoute = {
 97               context: context
 98             };
 99           }
100 
101         })
102 
103       })
104 
105     });
106 
107     statechart.initStatechart();
108 
109     monitor = statechart.get('monitor');
110     stateFoo = statechart.getState('foo');
111     stateBar = statechart.getState('bar');
112     stateA = statechart.getState('a');
113     stateB = statechart.getState('b');
114     stateC = statechart.getState('c');
115     stateX = statechart.getState('x');
116     stateY = statechart.getState('y');
117   },
118 
119   teardown: function() {
120     statechart = del = monitor = TestState = stateFoo = stateBar = stateA = stateB = stateC = stateX = stateY = null;
121   }
122 
123 });

156 test("trigger state B's route", function() {
157   var params = {};
158 
159   monitor.reset();
160 
161   ok(stateA.get('isCurrentState'), "state A should be a current state");
162   ok(!stateB.get('isCurrentState'), "state B should not be a current state");
163   ok(!stateC.get('isCurrentState'), "state C should not be a current state");
164   ok(stateX.get('isCurrentState'), "state X should be a current state");
165   ok(!stateY.get('isCurrentState'), "state Y should not be a current state");
166 
167   del.set('location', 'cat');
168 
169   stateB.routeTriggered(params);
170 
171   var seq = monitor.matchSequence().begin().exited('a').entered('b').end();
172   ok(seq, 'sequence should be exited[a], entered[b]');
173 
174   ok(!stateA.get('isCurrentState'), "state A should not be a current state");
175   ok(stateB.get('isCurrentState'), "state B should be a current state");
176   ok(!stateC.get('isCurrentState'), "state C should not be a current state");
177   ok(stateX.get('isCurrentState'), "state X should be a current state");
178   ok(!stateY.get('isCurrentState'), "state Y should not be a current state");
179 
180   var info = stateB.info;
181 
182   ok(info.enterState, "state B's enterState should have been invoked");
183 
184   var context = info.enterState.context;
185 
186   ok(SC.kindOf(context, SC.StateRouteHandlerContext), "state B's enterState method should have been provided a state route handler context object");
187   equals(context.get('state'), stateB);
188   equals(context.get('location'), 'cat');
189   equals(context.get('params'), params);
190   equals(context.get('handler'), stateB.routeTriggered);
191 });
